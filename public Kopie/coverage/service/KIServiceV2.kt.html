    private fun phaseTwo(){
       val actionMap : MutableMap<Action,Int> = mutableMapOf(
           move1Token to move1TokenScore() ,
           expandZoo to expandZooScore() ,
           discard1Token to discard1TokenScore(),
           purchase1Token to purchase1TokenScore(),
           swap2Types to swap2Types()
       )
        val bestAction =actionMap.maxByOrNull { it.value }
        if(checkNotNull(bestAction).value !=0){
            when(bestAction.key){
                Action.Discard1Token-> MoneyAction.Discard1Token(discard1Token.chosenAnimal)
                Action.Purchase1Token->MoneyAction.Purchase1Token(purchase1Token.moveLocation
                                                    ,purchase1Token.sPlayer,purchase1Token.chosenAnimal)
                Action.ExpandZoo ->MoneyAction.ExpandZoo
                Action.Swap2Types->MoneyAction.Swap2Types(swap2Types.map)
                Action.Move1Token -> MoneyAction.Move1Token(move1Token.moveLocation,
                                                                     move1Token.choseCard)
            }
        }

    private fun swap2Types(): Int {
        val currentGame=rootService.gameMaster.currentGame
        if(currentGame.currentPlayer.balance<1) return 0
        var score=0
        for(animal in currentGame.currentPlayer.zoo.barn.animals){
            val animalType= mutableListOf<Card.Animal>()
            for(tile in currentGame.currentPlayer.zoo.barn.animals){
                if(tile.animalType==animal.animalType){
                    animalType.add(tile)
                }
            }
            for(enclosure in currentGame.currentPlayer.zoo.enclosures){
                if(!enclosure.isBarn){
                    val enclosureType = if (enclosure.animals.isNotEmpty()) enclosure.animals.first().animalType
                                        else AnimalType.Panda
                    val enclosureSize=enclosure.animals.size
                    val animalNum=animalType.size
                    if(animalNum>enclosureSize){
                        if(animalNum>enclosure.animalCapacity){
                            if((enclosure.animalCapacity-enclosureSize)*2 > score){
                                score=(enclosure.animalCapacity-enclosureSize)*2
                               swap2Types.map = assignMap(currentGame.currentPlayer.zoo.barn,animal.animalType,
                               enclosure,enclosureType)
                            }else if(enclosureSize != enclosure.animalCapacity){
                                if((animalNum-enclosureSize)*2 > score){
                                    score=(animalNum-enclosureSize)*2
                                    swap2Types.map= assignMap(currentGame.currentPlayer.zoo.barn,animal.animalType,
                                        enclosure,enclosureType)
                                }
                            }
                        }
                    }
                }
            }
        }
        val fullSpace = currentGame.currentPlayer.zoo.enclosures.filter { !it.isBarn &&
        it.animals.size==it.animalCapacity}
        val notEmptySpaces=currentGame.currentPlayer.zoo.enclosures.filter { !it.isBarn &&
        it.animals.isNotEmpty()}
        for(full in fullSpace){
            for(notFull in notEmptySpaces){
                if(full.animalCapacity > notFull.animals.size && full.animalCapacity <= notFull.animalCapacity){
                    if(notFull.animalCapacity % 2 > score){
                        score = notFull.animalCapacity % 2
                        swap2Types.map= assignMap(notFull,notFull.animals.first().animalType,
                            full,full.animals.first().animalType)
                    }
                }
            }
        }
        return score
    }

    private fun assignMap(first: Enclosure, firstType: AnimalType,
                          second: Enclosure, secondType: AnimalType): Map<Card, Enclosure?> {
        val map : MutableMap<Card,Enclosure> = mutableMapOf()
        for (animal in first.animals){
            if(animal.animalType==firstType){
                map[animal] = second
            }
        }
        for (animal in second.animals){
            if(animal.animalType==secondType){
                map[animal] = first
            }
        }
        return map
    }

    private fun purchase1TokenScore(): Int {
        val currentGame =rootService.gameMaster.currentGame
        if(currentGame.currentPlayer.balance<2) return 0
        var score =0
        for (player in currentGame.players){
            if(player!=currentGame.currentPlayer){
                var bonus=0
                for(token in player.zoo.barn.animals){
                    val enclosuresHasAnimal = currentGame.currentPlayer.zoo.enclosures.filter { !it.isBarn &&
                            it.animals.first().animalType==token.animalType &&(it.animalCapacity>it.animals.size) }
                    for(e in enclosuresHasAnimal){
                        val size=e.animals.size
                        if (size+1==e.animalCapacity){
                            if(e.pointsFull> bonus){
                                bonus=e.pointsFull
                                move1Token.moveLocation = e
                                move1Token.sPlayer = player
                                move1Token.chosenAnimal =token
                            }
                        } else if (size+1>=e.pointsNearlyFull) {
                            if(e.pointsNearlyFull > bonus){
                                bonus=e.pointsNearlyFull
                                move1Token.moveLocation = e
                                move1Token.sPlayer = player
                                move1Token.chosenAnimal =token
                            }
                        } else {
                            if(2 > bonus){
                                bonus=2
                                move1Token.moveLocation = e
                                move1Token.sPlayer = player
                                move1Token.chosenAnimal =token
                            }
                        }
                    }
                }
                score =bonus
            }
        }
        return score
    }

    private fun discard1TokenScore(): Int {
        val currentGame=rootService.gameMaster.currentGame
        if(currentGame.currentPlayer.balance<2) return 0
        var bonus=0
        if(!hasEmptySpace(currentGame.currentPlayer.zoo.enclosures)
            && currentGame.currentPlayer.zoo.expandingValue()==0){
            var animalType : Card
            for(animal in currentGame.currentPlayer.zoo.barn.animals){
                var count =0
                animalType = animal
                for(tile in currentGame.currentPlayer.zoo.barn.animals) {
                    if (animal.animalType==tile.animalType) count++
                }
                if (count==1 ) {
                    if(!tradableAnimal(animal,currentGame)) {
                        if(4 > bonus ) {
                            bonus = 4
                            discard1Token.chosenAnimal= animalType
                        }
                    }
                }
            }
        }
        return bonus
    }


    private fun expandZooScore(): Int {
        val currentGame=rootService.gameMaster.currentGame
        if (currentGame.currentPlayer.balance<3 || currentGame.currentPlayer.zoo.expandingValue()==0) return 0
        val distinctAnimals =currentGame.currentPlayer.zoo.barn.animals
        var maxCount=0
        for (animal in distinctAnimals){
           var count =0
            for(tile in distinctAnimals) {
                if (animal.animalType==tile.animalType)count++
            }
            maxCount=max(maxCount,count)
        }
        return (maxCount * 2) + if (currentGame.currentPlayer.zoo.enclosures
                .none { it.shops.size != 0 }
        ) 0
        else 2
    }

    private fun move1TokenScore(): Int {
        val currentGame =rootService.gameMaster.currentGame
        if (currentGame.currentPlayer.balance<1) return 0
        var bonus=0
        for (token in currentGame.currentPlayer.zoo.barn.animals){
            val enclosuresHasAnimal = currentGame.currentPlayer.zoo.enclosures.filter {
                !it.isBarn && it.animals.first().animalType==token.animalType &&(it.animalCapacity>it.animals.size) }
            if (enclosuresHasAnimal.isNotEmpty()){
                for(e in enclosuresHasAnimal) {
                    val size = e.animals.size
                    if (size + 1 == e.animalCapacity && e.pointsFull > bonus ) {
                        bonus = e.pointsFull
                        move1Token.moveLocation = e
                        move1Token.choseCard = token
                    }
                    else if (size + 1 >= e.pointsNearlyFull && e.pointsNearlyFull > bonus){
                        bonus = e.pointsNearlyFull
                        move1Token.moveLocation = e
                        move1Token.choseCard = token
                    }
                    else {
                        if(2 > bonus){
                            bonus =2
                            move1Token.moveLocation = e
                            move1Token.choseCard = token
                        }
                    }
                }
            }else{
                if(hasEmptySpace(currentGame.currentPlayer.zoo.enclosures) && 3 > bonus){
                    bonus =3
                    move1Token.moveLocation = currentGame.currentPlayer.zoo.enclosures
                        .first { !it.isBarn && it.animals.isEmpty() }
                    move1Token.choseCard = token
                }
            }
        }
        for(token in currentGame.currentPlayer.zoo.barn.shops){
            val enclosures = currentGame.currentPlayer.zoo.enclosures.filter{
                !it.isBarn && it.shops.size<it.shopCapacity}
            for(e in enclosures){
                val size=e.shops.size
                if(size==0 && e.animals.size!=0 && 2+e.animals.size>bonus) {
                    bonus=2+e.animals.size
                    move1Token.moveLocation = e
                    move1Token.choseCard = token
                }
                else {
                    if(2 > bonus){
                        bonus = 2
                        move1Token.moveLocation = e
                        move1Token.choseCard = token
                    }
                }
            }
        }
        return bonus
    }

    private fun phaseOne(){
        val currentGame =rootService.gameMaster.currentGame
        val depth=rootService.gameMaster.currentGame.players.size
        val turn =maxiMax(rootService.gameMaster.currentGame,initializeDrawScore(),depth)
        if(turn.second==null){
            rootService.playerActionService.turn(Turn.Draw(tileOnTruck!!),currentGame)
        }else{
            rootService.playerActionService.turn(Turn.TakeTruck(turn.second!!,placeTiles(turn.second!!)),currentGame)
        }
    }

    private fun placeTiles(second: Truck): Map<Card, Enclosure?> {
        var position = 0
        val enclosuresMap =
            rootService.gameMaster.currentGame.currentPlayer.zoo.enclosures.toMutableList().map { it to position++ }
        return second.cards.associateWith {
            assignAnEnclosure(
                it,
                enclosuresMap,
                rootService.gameMaster.currentGame.currentPlayer.zoo
            )
        }
    }

    private fun assignAnEnclosure(card: Card, enclosures: List<Pair<Enclosure, Int>>, zoo: Zoo): Enclosure? {
        when (card) {
            is Card.Animal -> {
                enclosures.sortedByDescending { it.first.animals.size }
                for (enclosure in enclosures.filter { it.first.animals.isEmpty() ||
                        (it.first.animals.size<it.first.animalCapacity && it.first.animals.first().animalType==card.animalType )}){
                    enclosure.first.animals.add(card)
                    return zoo.enclosures[enclosure.second]
                }
                return null
            }
            is Card.Shop -> {
                enclosures.sortedByDescending { it.first.animals.size }
                for (enclosure in enclosures.filter { it.first.shops.size<it.first.shopCapacity }){
                    enclosure.first.shops.add(card)
                    return zoo.enclosures[enclosure.second]
                }
                return null
            }
            else -> return null
        }
    }

        val takeOneTruckScore = bestTruckScore(game)
        //depth
        if(game.currentPlayer.selectedTruck!=null || depth==0){
            bestPair= Pair(max(drawScore,takeOneTruckScore.first),null)

                        drawNPlace(drawGameState, tile, truck)

                        bestPair = if (game.currentPlayer == rootService.gameMaster.currentGame.currentPlayer) {

    private fun min(firstPair: Pair<Int, Truck?>, secondPair: Pair<Int, Truck?>): Pair<Int, Truck?> {
        return if(firstPair.first <= secondPair.first) firstPair
        else secondPair
    }

    private fun maxiMax(game: ZoolorettoGame?, drawScore:Int, depth:Int):Pair<Int, Truck?> {
        checkNotNull(game)
        var bestPair :Pair<Int,Truck?> = Pair(Int.MIN_VALUE,null)
        val takeOneTruckScore = bestTruckScore(game)

                        bestPair = if (game.currentPlayer == rootService.gameMaster.currentGame.currentPlayer) {
                            max(maxiMax(drawGameState, drawScore - 2,depth-1), takeOneTruckScore)
                        } else {
                            if (drawScore >= takeOneTruckScore.first) {
                                maxiMax(drawGameState, drawScore -2 ,depth-1)
                            } else {
                                maxiMax(truckGameState, drawScore,depth-1)
                            }
                        }
                    }
                }
            }
        }
        return bestPair
    }


    private fun max(firstPair: Pair<Int, Truck?>, secondPair: Pair<Int, Truck?>): Pair<Int, Truck?> {
        return if(firstPair.first>=secondPair.first) firstPair
        else secondPair
    }


    private fun takeTruck(second: Truck?, gameState: ZoolorettoGame) {
        gameState.currentPlayer.selectedTruck = second
        gameState.trucks.remove(second)
        gameState.currentPlayer = nextPlayer(gameState.currentPlayer)
    }

    private fun drawNPlace(gameState: ZoolorettoGame, tile: Card, truck: Truck) {
        truck.cards.add(tile)
        gameState.deck.remove(tile)
        gameState.currentPlayer = nextPlayer(gameState.currentPlayer)
    }

    private fun nextPlayer(player: Player): Player {
        val playerNumber=rootService.gameMaster.currentGame.players.size
        val playerIndex = rootService.gameMaster.currentGame.players.indexOf(player)
        return rootService.gameMaster.currentGame.players[(playerIndex+1)%playerNumber]
        }

    private fun allHasTruck(game: ZoolorettoGame): Boolean {
        for(player in game.players){
            if(player.selectedTruck==null)
                return false
        }
        return true
    }

    private fun bestTruckScore(game: ZoolorettoGame): Pair<Int,Truck?> {
        var bestTruck:Pair<Int,Truck?> = Pair(-6,null)
        for(truck in game.trucks){
            var score =-truck.capacity*2
            for(tile in truck.cards){
                var bonus=0
                when (tile) {
                    is Card.Animal -> {
                        val enclosures = hasAnimal(tile, game.currentPlayer.zoo.enclosures)
                        enclosures.removeIf { it.first.animalCapacity == it.first.animals.size }
                        if (enclosures.isNotEmpty()) {
                            for (enclosure in enclosures) {
                                bonus = if (enclosure.first.animals.size + 1 == enclosure.first.pointsNearlyFull) {
                                    enclosure.first.pointsNearlyFull
                                } else if (enclosure.first.animals.size + 1 == enclosure.first.pointsFull) {
                                    enclosure.first.pointsFull
                                } else {
                                    if(enclosure.first.shops.size!=0){
                                        enclosure.first.animals.size+1
                                    }else{
                                        1
                                    }
                                }
                            }
                        } else {
                            bonus = if (hasEmptySpace(game.currentPlayer.zoo.enclosures)) {
                                3
                            } else if (tradableAnimal(tile,game)) {
                                3
                            } else {
                                2
                            }
                        }
                    }
                    is Card.Shop -> {
                        bonus = if(hasEmptySpaceTwo(game.currentPlayer.zoo.enclosures)){
                            if (!hasShop(tile,game.currentPlayer.zoo.enclosures)){
                                4
                            } else 3
                        }else{
                            if(tradeableShop(tile,game)) 4
                            else 0
                        }
                    }
                    else -> bonus=4
                }
                score +=bonus
            }
            if(score > bestTruck.first) bestTruck=Pair(score,truck)
        }
        return bestTruck
    }

    private fun tradeableShop(tile: Card, game: ZoolorettoGame): Boolean {
        for(player in game.players){
            if (player!=game.currentPlayer){
                for(enclosure in player.zoo.enclosures){
                    if(!enclosure.isBarn){
                        var hasEmptySlot=false
                        var newShopType =true
                        if(enclosure.shopCapacity>enclosure.shops.size) hasEmptySlot=true
                        if (enclosure.shops.isNotEmpty() && enclosure.shops.contains(tile)) newShopType=false
                        if(hasEmptySlot&&newShopType) return true
                    }
                }
            }
        }
        return false
    }

    private fun hasShop(tile: Card.Shop, enclosures: MutableList<Enclosure>):Boolean {
        for(enclosure in enclosures) {
            if (!enclosure.isBarn) {
                for (shop in enclosure.shops) {
                    if (shop.shopType == tile.shopType) return true
                }
            }
        }
        return false
    }

    private fun hasEmptySpaceTwo(enclosures: MutableList<Enclosure>): Boolean {
        for(enclosure in enclosures){
            if (!enclosure.isBarn && enclosure.shops.isEmpty()) return true
        }
        return false
    }

    private fun tradableAnimal(tile: Card.Animal, game: ZoolorettoGame): Boolean {
        for(player in game.players){
            if (player!=game.currentPlayer){
                for(enclosure in player.zoo.enclosures){
                    if(!enclosure.isBarn && enclosure.animals.isNotEmpty()
                        && enclosure.animals.first().animalType==tile.animalType
                        && enclosure.animals.size+1<=enclosure.animalCapacity)
                        return true
                }
            }
        }
        return false
    }

    private fun hasEmptySpace(enclosures: MutableList<Enclosure>): Boolean {
        for(enclosure in enclosures){
            if (!enclosure.isBarn && enclosure.animals.isEmpty()) return true
        }
        return false
    }

    private fun hasAnimal(tile: Card.Animal, enclosures: MutableList<Enclosure>): MutableList<Pair<Enclosure,Int>> {
        val list= mutableListOf<Pair<Enclosure,Int>>()
        for(enclosure in enclosures) {
            if (!enclosure.isBarn) {
                val count = enclosure.animals.count { it.animalType == tile.animalType }
                if (count != 0) list.add(Pair(enclosure, count))
            }
        }
        return list
    }

    private fun initializeDrawScore(): Int {
        var score=0
        for(truck in rootService.gameMaster.currentGame.trucks){
            score+=truck.capacity*2
        }
        return score
    }
